    #include <stdio.h>
    #include <freertos/FreeRTOS.h>
    #include <freertos/task.h>
    #include <driver/gpio.h> 

    #define I_ban 2 //led incluido
    #define F_ban 4
    #define NFC 5
    #define LED 22

    void app_main(void){
        
        gpio_pad_select_gpio(I_ban);
        gpio_set_direction(I_ban,GPIO_MODE_INPUT);
        gpio_pad_select_gpio(F_ban);
        gpio_set_direction(F_ban,GPIO_MODE_INPUT);
        gpio_pad_select_gpio(NFC);
        gpio_set_direction(NFC,GPIO_MODE_INPUT);

        for(;;){
            gpio_get_level(I_ban);
            vTaskDelay(100/portTICK_PERIOD_MS);
            gpio_get_level(F_ban);
            vTaskDelay(100/portTICK_PERIOD_MS);

            if (I_ban==1){
                gpio_set_level(22,1);
            }
            if (F_ban==1){
                gpio_set_level(LED,1);
            }

        }
    }
